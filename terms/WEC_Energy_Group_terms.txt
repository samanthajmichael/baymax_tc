<script>
    // script portions by Gregg Buntin - WEC Energy Group

    // this is called to get the stock data
    function getStockQuote() 
    {
        // setup for our stock quote
        var data =
        {
            apiKey: "BF185719B0464B3CB809D23926182246",
            exchange: "NYSE",
            symbol: "WEC",
            pageSize: 1
        } // var data

        // get the data
        $.ajax({
            type: "GET",
            url: "https://wecenergy2018ir.q4web.com/feed/StockQuote.svc/GetStockQuoteList",
            data: data,
            dataType: "json",
            error: stockQuoteError,
            success: gotStockQuote
        });
    } // function getStockQuote
    
    // this is called if there is an error getting the stock quote
    function stockQuoteError() {
        $("#stockQuoteDiv").hide(); // since there was an error lets hide the stock quote rather than show invalid data
    } // function stockQuoteError
    
    // this is called when the data arrives
    function gotStockQuote(inStockData) {
        var stockTrade = "";
        var stockChange = "";
        
        // the data comes in as JSON data so we need to parse it out
        try {
            stockTrade = inStockData.GetStockQuoteListResult[0].TradePrice;
            stockChange = inStockData.GetStockQuoteListResult[0].Change;
        } catch(err) {
        } // try getting the data from the JSON object
        
        var stockUpDown = "";
        if (stockChange > 0) {
            stockUpDown = "+";
        } // if stock change is positive (negative should already have the minus)
        
        // place the data into the spans
        try {
            $("#stockQuoteTrade").html("$" + stockTrade);
            $("#stockQuoteChange").html(stockUpDown + stockChange);
        } catch(err) {} // try placing the data into the spans
    } // function gotStockQuote
    
    $("body").ready(function() { getStockQuote(); });
</script>
